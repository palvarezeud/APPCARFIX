### Manejo de Errores Global con `IExceptionHandler` (.NET 10)

.NET 10 incluye `IExceptionHandler` y `ProblemDetails` de forma nativa. No se necesita middleware manual.

```csharp
// CarFix.API/Middleware/ExceptionHandlingMiddleware.cs
public class ExceptionHandlingMiddleware
// CarFix.API/Exceptions/GlobalExceptionHandler.cs
public class GlobalExceptionHandler : IExceptionHandler
{
    private readonly RequestDelegate _next;
    private readonly ILogger<ExceptionHandlingMiddleware> _logger;
    private readonly ILogger<GlobalExceptionHandler> _logger;

    public ExceptionHandlingMiddleware(RequestDelegate next, ILogger<ExceptionHandlingMiddleware> logger)
    {
        _next   = next;
        _logger = logger;
    }
    public GlobalExceptionHandler(ILogger<GlobalExceptionHandler> logger)
        => _logger = logger;

    public async Task InvokeAsync(HttpContext ctx)
    public async ValueTask<bool> TryHandleAsync(
        HttpContext ctx,
        Exception   exception,
        CancellationToken ct)
    {
        try
        var (statusCode, title, detail) = exception switch
        {
            await _next(ctx);
        }
        catch (ValidationException ex)
            ValidationException ex => (
                StatusCodes.Status400BadRequest,
                "Validation error",
                string.Join("; ", ex.Errors.Select(e => e.ErrorMessage))),

            DomainException ex => (
                StatusCodes.Status422UnprocessableEntity,
                "Domain rule violation",
                ex.Message),

            _ => (
                StatusCodes.Status500InternalServerError,
                "Internal server error",
                "An unexpected error occurred.")
        };

        if (statusCode == 500)
            _logger.LogError(exception, "Unhandled exception");

        ctx.Response.StatusCode = statusCode;
        await ctx.Response.WriteAsJsonAsync(new ProblemDetails
        {
            ctx.Response.StatusCode = StatusCodes.Status400BadRequest;
            await ctx.Response.WriteAsJsonAsync(new
            {
                Errors = ex.Errors.Select(e => new { e.PropertyName, e.ErrorMessage })
            });
        }
        catch (DomainException ex)
        {
            ctx.Response.StatusCode = StatusCodes.Status422UnprocessableEntity;
            await ctx.Response.WriteAsJsonAsync(new { Error = ex.Message });
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Error no controlado");
            ctx.Response.StatusCode = StatusCodes.Status500InternalServerError;
            await ctx.Response.WriteAsJsonAsync(new { Error = "Error interno del servidor." });
        }
            Status = statusCode,
            Title  = title,
            Detail = detail
        }, ct);

        return true;
    }
}
```

## 0. Convenciones de Nomenclatura

**Todo el código se escribe en español, sin caracteres especiales.**

- Sin tildes: `fecha` no `fécha`, `numero` no `número`
- Sin ñ: sustituir por `nn` seguido de la vocal — ejemplo: `annio` en lugar de `año`, `espannol` en lugar de `español`
- Los sufijos de patrones técnicos reconocidos pueden quedar en inglés: `Repository`, `Handler`, `Validator`, `Command`, `Query`, `Dto`, `Configuration`, `Exception`

| Elemento | Convención | Ejemplo |
|---|---|---|

| Métodos | PascalCase español | `ObtenerPorIdAsync`, `AgregarAsync` |
| Propiedades | PascalCase español | `EsExitoso`, `FechaIngreso` |

// CarFix.Application/Common/Behaviors/ValidationBehavior.cs
public class ValidationBehavior<TRequest, TResponse> : IPipelineBehavior<TRequest, TResponse>
    where TRequest : notnull
// CarFix.Application/Comun/Comportamientos/ComportamientoValidacion.cs
public class ComportamientoValidacion<TSolicitud, TRespuesta> : IPipelineBehavior<TSolicitud, TRespuesta>
    where TSolicitud : notnull
{
    private readonly IEnumerable<IValidator<TRequest>> _validators;
    private readonly IEnumerable<IValidator<TSolicitud>> _validadores;

    public ValidationBehavior(IEnumerable<IValidator<TRequest>> validators)
        => _validators = validators;
    public ComportamientoValidacion(IEnumerable<IValidator<TSolicitud>> validadores)
        => _validadores = validadores;

    public async Task<TResponse> Handle(TRequest request, RequestHandlerDelegate<TResponse> next, CancellationToken ct)
    public async Task<TRespuesta> Handle(TSolicitud solicitud, RequestHandlerDelegate<TRespuesta> siguiente, CancellationToken ct)
    {
        if (!_validators.Any()) return await next();
        if (!_validadores.Any()) return await siguiente();

        var context = new ValidationContext<TRequest>(request);
        var failures = _validators
            .Select(v => v.Validate(context))
        var contexto = new ValidationContext<TSolicitud>(solicitud);
        var fallas = _validadores
            .Select(v => v.Validate(contexto))
            .SelectMany(r => r.Errors)
            .Where(f => f is not null)
            .ToList();

        if (failures.Any())
            throw new ValidationException(failures);
        if (fallas.Any())
            throw new ValidationException(fallas);

        return await next();
        return await siguiente();
    }
}

```csharp
// CarFix.Application/Comun/Comportamientos/ComportamientoValidacion.cs
public class ComportamientoValidacion<TSolicitud, TRespuesta> : IPipelineBehavior<TSolicitud, TRespuesta>
    where TSolicitud : notnull
{
    private readonly IEnumerable<IValidator<TSolicitud>> _validadores;

    public ComportamientoValidacion(IEnumerable<IValidator<TSolicitud>> validadores)
        => _validadores = validadores;

    public async Task<TRespuesta> Handle(TSolicitud solicitud, RequestHandlerDelegate<TRespuesta> siguiente, CancellationToken ct)
    {
        if (!_validadores.Any()) return await siguiente();

        var contexto = new ValidationContext<TSolicitud>(solicitud);
        var fallas = _validadores
            .Select(v => v.Validate(contexto))
            .SelectMany(r => r.Errors)
            .Where(f => f is not null)
            .ToList();

        if (fallas.Any())
            throw new ValidationException(fallas);

        return await siguiente();
    }
}

public interface IServicioToken
{
    string GenerarToken(int usuarioId, string nombreUsuario, string rol);
}

// CarFix.Infrastructure/Seguridad/ServicioToken.cs
public class ServicioToken : IServicioToken
{
    private readonly IConfiguration _config;

    public ServicioToken(IConfiguration config) => _config = config;

    public string GenerarToken(int usuarioId, string nombreUsuario, string rol)
    {
        var llave     = new SymmetricSecurityKey(Encoding.UTF8.GetBytes(_config["Jwt:Llave"]!));
        var credenciales = new SigningCredentials(llave, SecurityAlgorithms.HmacSha256);

        var reclamaciones = new[]
        {
            new Claim(ClaimTypes.NameIdentifier, usuarioId.ToString()),
            new Claim(ClaimTypes.Name,            nombreUsuario),
            new Claim(ClaimTypes.Role,            rol)
        };

        var token = new JwtSecurityToken(
            issuer:             _config["Jwt:Emisor"],
            audience:           _config["Jwt:Audiencia"],
            claims:             reclamaciones,
            expires:            DateTime.UtcNow.AddMinutes(double.Parse(_config["Jwt:ExpiracionMinutos"]!)),
            signingCredentials: credenciales
        );

        return new JwtSecurityTokenHandler().WriteToken(token);
    }
}

### Evitar (Anti-patrones)

| Anti-patron | Correccion |
|---|---|
| Logica de negocio en endpoints | Mover al Handler o a la entidad de dominio |
| Entidades anemicas con solo getters/setters publicos | Encapsular comportamiento con metodos de dominio |
| Repositorio generico `IRepositorio<T>` con CRUD global | Repositorios especificos con metodos con nombre de dominio |
| `DbContext` inyectado directo en handlers de commands | Usar `IUnidadTrabajo` en commands; `ICarFixDbContext` en queries |
| `try/catch` para controlar flujo de negocio | Usar `Resultado<T>` para errores esperados; excepciones solo para lo verdaderamente excepcional |
| DTOs que exponen el modelo de dominio directamente | Mapear siempre a DTOs especificos por caso de uso |
| Toda la logica en un unico proyecto | Respetar la separacion en 4 proyectos/capas |
| Editar archivos en `Persistencia/Generado/` | Usar clases parciales en `Persistencia/Extensiones/` |
| Crear entidades a mano (Code First) | Siempre usar scaffold desde la BD |
| Logica de negocio en endpoints | Mover al Handler o a la clase parcial de la entidad |
| Exponer el DbContext fuera de Infrastructure | Inyectar `ICarFixDbContext` (interfaz) en queries |
| Guardar la llave JWT en `appsettings.json` en produccion | Usar variables de entorno o secrets manager |
| Endpoints sin `RequireAuthorization()` | Todo endpoint (excepto login) debe estar protegido |

> Cómo se hacen las especificaciones en spec driven development?

Spec Driven Development (SDD) parte de escribir la especificación del comportamiento antes de escribir el código. La especificación es ejecutable — actúa como documentación viva y como suite de pruebas al mismo tiempo.

---
Formato base: Given / When / Then

Cada especificación describe un escenario de negocio con tres partes:

Feature: Crear orden de reparacion

  Scenario: El mecanico crea una orden para un vehiculo registrado
    Given que existe un vehiculo con ID 5 registrado en el sistema
    When el mecanico crea una orden con el problema "Ruido en el motor"
    Then la orden queda en estado "Recibido"
    And el total de la orden es 0

  Scenario: No se puede crear una orden sin describir el problema
    Given que existe un vehiculo con ID 5 registrado en el sistema
    When el mecanico intenta crear una orden sin descripcion de problema
    Then el sistema rechaza la solicitud con el mensaje "Debe describir el problema"

---
El proceso

1. Escribir el Feature (spec legible para negocio y desarrollo)
        ↓
2. Ejecutar → falla (no existe código aún)
        ↓
3. Implementar el código mínimo para que pase
        ↓
4. Refactorizar
        ↓
5. Repetir con el siguiente escenario

La diferencia con TDD es el nivel de abstracción: TDD trabaja a nivel de unidad de código; SDD trabaja a nivel de comportamiento de negocio.

---
En .NET 10 la herramienta es SpecFlow (o Reqnroll)

▎ Nota: SpecFlow fue discontinuado en 2024. Su fork activo y compatible con .NET 10 se llama Reqnroll y es el reemplazo directo.

dotnet add package Reqnroll.Microsoft.Extensions.DependencyInjection

## 8. Especificaciones — Spec Driven Development (Reqnroll)

Las especificaciones describen el comportamiento del sistema en lenguaje de negocio **antes** de escribir el codigo. Cada spec es un escenario ejecutable que actua como documentacion viva y prueba de aceptacion al mismo
 tiempo.

> **Herramienta:** Reqnroll (fork activo de SpecFlow, compatible con .NET 10). No usar SpecFlow — fue discontinuado en 2024.

### Proyecto y paquetes

```bash
# Crear el proyecto de especificaciones dentro de la carpeta Especificaciones/
dotnet new nunit -n CarFix.Especificaciones -o Especificaciones/CarFix.Especificaciones
dotnet add Especificaciones/CarFix.Especificaciones package Reqnroll.NUnit
dotnet add Especificaciones/CarFix.Especificaciones package Reqnroll.Microsoft.Extensions.DependencyInjection
dotnet add Especificaciones/CarFix.Especificaciones package Microsoft.EntityFrameworkCore.InMemory
```

### Alcance

Las specs cubren la **capa de Aplicacion** (handlers de MediatR). Prueban las reglas de negocio sin depender del transporte HTTP. La BD se reemplaza por un proveedor en memoria (`UseInMemoryDatabase`) para que las prueb
as sean rapidas y sin efectos secundarios.

### Convencion de archivos `.feature`

- Un archivo `.feature` por entidad o caso de uso principal
- Nombre en espannol, sin caracteres especiales: `OrdenesReparacion.feature`
- Cada `Scenario` describe un caso de negocio completo con datos concretos
- Usar `Scenario Outline` + `Examples` cuando el mismo flujo cambia solo los datos

### Ejemplo completo — Ordenes de Reparacion

```gherkin

### Step Definitions

```csharp
// Especificaciones/CarFix.Especificaciones/Steps/OrdenReparacionSteps.cs
[Binding]
public class OrdenReparacionSteps
{
    private readonly ISender          _sender;
    private readonly ICarFixDbContext _contexto;
    private Resultado<int>            _ultimoResultado;
    private Resultado                 _ultimoResultadoSimple;

    public OrdenReparacionSteps(ISender sender, ICarFixDbContext contexto)
    {
        _sender   = sender;
        _contexto = contexto;
    }

    [Given(@"que existe un vehiculo con ID (\d+) registrado en el sistema")]
    public async Task DadoExisteVehiculo(int vehiculoId)
    {
        _contexto.Vehiculos.Add(new Vehiculo
        {
            VehiculoID        = vehiculoId,
            Marca             = "Toyota",
            EsAutomatico      = false,
            DetallesCarroceria = "Sedan"
        });
        await _contexto.GuardarCambiosAsync();
    }

    [When(@"el mecanico crea una orden para el vehiculo (\d+) con el problema ""(.*)""")]
    public async Task CuandoCreaOrden(int vehiculoId, string problema)
    {
        _ultimoResultado = await _sender.Send(new CrearOrdenCommand(vehiculoId, problema));
    }

    [When(@"el mecanico intenta crear una orden para el vehiculo (\d+) sin descripcion")]
    public async Task CuandoCreaOrdenSinProblema(int vehiculoId)
    {
        _ultimoResultado = await _sender.Send(new CrearOrdenCommand(vehiculoId, string.Empty));
    }

    [Then(@"la orden se crea exitosamente")]
    public void EntoncesOrdenCreada()
        => Assert.That(_ultimoResultado.EsExitoso, Is.True);

    [Then(@"la orden queda en estado ""(.*)""")]
    public async Task EntoncesEstado(string estado)
    {
        var orden = await _contexto.OrdenesReparacion
            .Include(o => o.EstadoOrden)
            .FirstAsync(o => o.OrdenReparacionID == _ultimoResultado.Valor);

        Assert.That(orden.EstadoOrden.Descripcion, Is.EqualTo(estado));
    }

    [Then(@"el sistema rechaza la solicitud con el mensaje ""(.*)""")]
    public void EntoncesRechazo(string mensaje)
    {
        Assert.That(_ultimoResultado.EsExitoso, Is.False);
        Assert.That(_ultimoResultado.Error, Is.EqualTo(mensaje));
    }
}

## 3. Guia de Arquitectura Limpia — Backend ASP.NET Core

### Stack confirmado por el usuario
- **Framework:** ASP.NET Core Web API
- **Version:** .NET 10
- **Lenguaje:** C#
- **ORM:** Entity Framework Core — **Database First**
- **Autenticacion:** JSON Web Token (JWT)
- **Pruebas:** Spec Driven Development con Reqnroll

### Archivo creado
`.claude/arquitectura_limpia_backend.md`

### Decisiones de arquitectura tomadas en esta sesion

#### Estructura de la solucion (5 proyectos en 5 carpetas fisicas)
```
Backend/
├── CarFix.sln
├── Dominio/           → CarFix.Dominio
├── Aplicacion/        → CarFix.Aplicacion
├── Infraestructura/   → CarFix.Infraestructura
├── WebApi/            → CarFix.WebApi
└── Especificaciones/  → CarFix.Especificaciones
```

#### Patrones incluidos
- **Clean Architecture**: dependencias unidireccionales hacia adentro
- **CQRS con MediatR**: Commands (escritura) / Queries (lectura directa al DbContext)
- **Patron Resultado**: `Resultado<T>` con `EsExitoso`, `Valor`, `Error`, `Exito()`, `Fallo()`
- **Unidad de Trabajo**: `IUnidadTrabajo` / `UnidadTrabajo` con `GuardarCambiosAsync()`
- **Inversion de Control**: capas internas definen interfaces, Infrastructure implementa
- **Inyeccion de Dependencias**: cada capa tiene `InjeccionDependencias.cs`
- **Repositorios especificos**: metodos con nombres de dominio en espannol
- **Validacion automatica**: `ComportamientoValidacion` pipeline de MediatR + FluentValidation

#### Entity Framework Database First
- Entidades generadas con `dotnet ef dbcontext scaffold`
- Archivos generados en `Infraestructura/Persistencia/Generado/` — nunca editar
- Comportamiento de dominio agregado via clases parciales en `Infraestructura/Persistencia/Extensiones/`
- Nombres de entidades y propiedades identicos a los de la BD

#### JWT
- Paquete: `Microsoft.AspNetCore.Authentication.JwtBearer`
- `ServicioToken` en `Infraestructura/Seguridad/`
- Todos los endpoints protegidos con `.RequireAuthorization()`
- Unico endpoint publico: `POST /api/autenticacion/iniciar-sesion`
- Llave secreta nunca en repositorio — usar `dotnet user-secrets`

#### .NET 10 — APIs especificas
- `TypedResults` (no `Results`) para inferencia de tipos en OpenAPI
- `IExceptionHandler` + `AddExceptionHandler<ManejadorExcepciones>()`
- `AddOpenApi()` + `MapOpenApi()` — sin Swashbuckle
- `AddProblemDetails()` nativo

#### Spec Driven Development con Reqnroll
- Proyecto `Especificaciones/CarFix.Especificaciones`
- Herramienta: Reqnroll (reemplazo activo de SpecFlow para .NET 10)
- Specs cubren la capa de Aplicacion (handlers de MediatR)
- BD en memoria (`UseInMemoryDatabase`) — cada escenario parte limpio
- Formato Gherkin: `Feature` / `Background` / `Scenario` / `Scenario Outline`
- Flujo: escribir `.feature` → implementar steps → implementar handler → spec pasa
